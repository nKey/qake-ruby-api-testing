def create
  @score = Score.create!(
    player: params[:player],
    score: params[:score],
    time: DateTime.iso8601(params[:time])
  )
  json_response(@score, :created)
end

def index
  @scores = Score.by_players(@players)
                 .after(@after)
                 .before(@before)
                 .page(@page)
                 .per(@size)
                 .order(score: :desc, time: :desc)
  json_response({ list: @scores, page: @page, size: @scores.length }, :ok)
end
